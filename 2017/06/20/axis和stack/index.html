stack、axis

test_1 = np.array([[1, 2, 3], [4, 5, 6]])
test_2 = np.array([[11, 12, 13], [14, 15, 16]])

stack针对维度相同的矩阵合并

# axis=0,1,2分别代表所有，行合并，列合并
print(np.stack((test_1, test_2), axis=2))
print(np.stack((test_1, test_2), axis=1))
print(np.stack((test_1, test_2), axis=0))

axis=2,1,0时的结果分别是

print(np.stack((test_1, test_2), axis=2))
[[[ 1 11]
  [ 2 12]
  [ 3 13]]

 [[ 4 14]
  [ 5 15]
  [ 6 16]]]

#
print(np.stack((test_1, test_2), axis=1))
[[[ 1 2 3]
[11 12 13]]

[[ 4  5  6]
 [14 15 16]]]

#
print(np.stack((test_1, test_2), axis=0))
[[[ 1 2 3]
[ 4 5 6]]

[[11 12 13]
 [14 15 16]]]

hstack和vstack均针对于列向量和矩阵

# reshape在Numpy高级应用中介绍
# 类似于np.stack(*****, axis=2)
t = np.arange(1, 3).reshape((-1, 1))
print(np.hstack((t, test_1 )))

#
结果：
[[1 1 2 3]
[2 4 5 6]]
#

# 类似于np.stack(*****, axis=1)
t = np.arange(1, 4).reshape((1, -1))
print(np.vstack((t, test_1)))

#
结果：
[[1 2 3]
[1 2 3]
[4 5 6]]
