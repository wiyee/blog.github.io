public int compare(Entry<String, Integer> o1, Entry<String, Integer> o2) {
    return o2.getValue().compareTo(o1.getValue());
    }
});

for(Map.Entry<String,Integer> mapping:list){
hotGameMap.put(mapping.getKey(), mapping.getValue());
}
